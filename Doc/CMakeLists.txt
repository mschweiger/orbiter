# Orbiter.pdf ------------------------------------------------------------

odt_to_pdf_arglist("Orbiter" arglist src out)
add_custom_command(
	OUTPUT ${out}
	COMMAND ${ODT_TO_PDF_COMPILER} ${arglist}
	DEPENDS ${src}
	JOB_POOL soffice
)
add_custom_target(OrbiterDoc
	DEPENDS ${out}
)
add_dependencies(${OrbiterTgt}
	OrbiterDoc
)
set_target_properties(OrbiterDoc
	PROPERTIES
	FOLDER Doc
)
install(FILES ${out}
	DESTINATION ${ORBITER_INSTALL_DOC_DIR}
)
